def display_random_samples(data, labels, class_names, num_samples=25, figsize=(10, 10)):
    # Generate random indices
    random_indices = np.random.choice(len(data), num_samples, replace=False)
    
    # Create subplots
    rows = int(np.sqrt(num_samples))
    cols = int(np.ceil(num_samples / rows))
    fig, axes = plt.subplots(rows, cols, figsize=figsize)
    
    # Plot each sample
    for i, idx in enumerate(random_indices):
        row = i // cols
        col = i % cols
        ax = axes[row, col] if rows > 1 else axes[col]
        # Reshape the flattened image data to its original shape (28x28)
        image = data[idx].reshape(28, 28)
        ax.imshow(image, cmap='gray')
        ax.set_title(class_names[labels[idx]])
        ax.axis('off')
    
    plt.tight_layout()
    plt.show()

# Display random samples
display_random_samples(train_images, train_labels, class_names)

desired_class = 5

# Compute precision for class '5' using original predictions and true labels

# Set a higher threshold for predicting class '5'

# Create adjusted predictions based on the threshold

# Compute precision for class '5' using adjusted predictions and true labels

# Set a lower threshold for predicting class '5' (to capture more positive instances)
threshold = 0.3  # Adjust the threshold as needed

# Create adjusted predictions based on the threshold
adjusted_predictions = predictions[:, desired_class] >= threshold

# Compute recall for class '5' using adjusted predictions and true labels
adjusted_recall = recall_score(test_labels == desired_class, adjusted_predictions)
